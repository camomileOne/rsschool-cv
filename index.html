int getSumOfDivisors(int n) {
    int sum = 1;
    int i = 2;
    for (; i < n / i; i++) {
        if (n / i * i == n) {
            sum += i + n / i;
        }
    }
    if (i * i == n)
        sum += i;
    return sum;
}
